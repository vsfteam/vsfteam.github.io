311static inline uint8_t vsf_spi_get_data_bytes_from_mode(vsf_spi_mode_t mode)
312{
313 uint8_t bits = (mode & SPI_DATASIZE_MASK);
314 if (bits <= SPI_DATASIZE_8) {
315 return 1;
316 } else if (bits <= SPI_DATASIZE_16) {
317 return 2;
318 } else {
319 return 4;
320 }
321}
