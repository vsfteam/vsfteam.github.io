47typedef enum vsf_flash_irq_mask_t {
48 // 16: EOPIE(16) in FLASH_IER
49 VSF_FLASH_IRQ_ERASE_MASK = (1 << 16),
50 // 15: EOPIE(16) in FLASH_IER, shift right by 1 to avoid conflict with VSF_FLASH_IRQ_ERASE_MASK
51 VSF_FLASH_IRQ_WRITE_MASK = (1 << 16) >> 1,
52 // 14: implemented by software
53 VSF_FLASH_IRQ_READ_MASK = (1 << 14),
54
55 // 17..19, 21: WRPERRIE(17), PGSERRIE(18), STRBERRIE(19), INCERRIE(21) in FLASH_IER
56 VSF_FLASH_IRQ_WRITE_ERROR_MASK = (1 << 17) | (1 << 18) | (1 << 19) | (1 << 21),
57 // 24: RDSERRIE(24) in FLASH_IER
58 VSF_FLASH_IRQ_READ_ERROR_MASK = (1 << 24),
59
60 // not supported
61 // 22: OPERRIE(22) in FLASH_IER, not exist in STM32H7RS
62 VSF_FLASH_IRQ_ERASE_ERROR_MASK = (1 << 22),
