74static vsf_err_t __vsf_linux_httpd_urihandler_websocket_init(vsf_linux_httpd_request_t *req, uint8_t *data, uint_fast32_t size)
75{
76 VSF_LINUX_ASSERT((req != NULL) && (req->uri != NULL));
77 if (!req->websocket || (NULL == req->websocket_key)) {
78 req->response = VSF_LINUX_HTTPD_BAD_REQUEST;
79 return VSF_ERR_FAIL;
80 }
81
82 vsf_linux_httpd_urihandler_websocket_t *urihandler_websocket = &req->urihandler_ctx.websocket;
83 vsf_fifo_stream_t *stream;
84 int bufsize = sizeof(req->buffer) / 2;
85
86 urihandler_websocket->connected = false;
87
88 stream = &urihandler_websocket->stream_out;
89 stream->op = &vsf_fifo_stream_op;
90 stream->buffer = req->buffer;
91 stream->size = bufsize;
92 VSF_STREAM_INIT(stream);
93 stream->tx.param = req;
94 stream->tx.evthandler = __vsf_linux_httpd_urihandler_socket_stream_evthandler;
95 VSF_STREAM_CONNECT_TX(stream);
96 req->is_stream_out_started = true;
97 req->stream_out = &stream->use_as__vsf_stream_t;
98
99 stream = &urihandler_websocket->stream_in;
100 stream->op = &vsf_fifo_stream_op;
101 stream->buffer = req->buffer + bufsize;
102 stream->size = bufsize;
103 VSF_STREAM_INIT(stream);
104 stream->rx.param = req;
105 stream->rx.evthandler = __vsf_linux_httpd_urihandler_socket_stream_evthandler;
106 VSF_STREAM_CONNECT_RX(stream);
107 req->stream_in = &stream->use_as__vsf_stream_t;
108
109 vsf_stream_write_str(req->stream_out, "HTTP/1.1 101 Switching Protocols\r\nConnection: Upgrade\r\nUpgrade: websocket\r\nSec-WebSocket-Accept: ");
110
111 char secbuf[32 + sizeof(__websocket_magic)];
112 strcpy(secbuf, req->websocket_key);
113 strcat(secbuf, __websocket_magic);
114 free(req->websocket_key);
115 req->websocket_key = NULL;
116
117 mbedtls_sha1_context ctx;
118 uint8_t sha1_output[20];
119 mbedtls_sha1_init(&ctx);
120 mbedtls_sha1_starts(&ctx);
121 mbedtls_sha1_update(&ctx, (const unsigned char *)secbuf, strlen(secbuf));
122 mbedtls_sha1_finish(&ctx, sha1_output);
123 mbedtls_sha1_free(&ctx);
124
125 size_t olen;
126 mbedtls_base64_encode((unsigned char *)secbuf, sizeof(secbuf), &olen,
127 (const unsigned char *)sha1_output, sizeof(sha1_output));
128 vsf_stream_write_str(req->stream_out, secbuf);
129 vsf_stream_write_str(req->stream_out, "\r\n\r\n");
130
131 req->response = VSF_LINUX_HTTPD_OK;
132 return VSF_ERR_NONE;
133}
