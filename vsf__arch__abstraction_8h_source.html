303 VSF_MACRO_SAFE_NAME(local_cnt) = 1; \
304 (((__region) != NULL) && VSF_MACRO_SAFE_NAME(local_cnt)--) ? \
305 ((VSF_MACRO_SAFE_NAME(protect_status) = (__region)->enter()), true) : false;\
