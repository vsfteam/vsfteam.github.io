94static inline uint32_t __hw_io_reg_read(bool is_pmic, volatile uint32_t *reg)
95{
96 if (!is_pmic) {
97 return *reg;
98 } else {
99 return PMIC_MEM_READ((unsigned int)reg);
100 }
101}
102
103static inline void __hw_io_reg_mask_write(bool is_pmic, volatile uint32_t *reg,
104 uint32_t wdata, uint32_t wmask)
105{
106 if (!is_pmic) {
107 *reg = (*reg & ~wmask) | (wdata & wmask);
108 } else {
109 PMIC_MEM_MASK_WRITE((unsigned int)reg, wdata, wmask);
110 }
111}
