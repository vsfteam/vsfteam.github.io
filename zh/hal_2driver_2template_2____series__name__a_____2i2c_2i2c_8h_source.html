61/*\note hw I2C driver can reimplement vsf_i2c_mode_t/vsf_i2c_cmd_t/vsf_i2c_irq_mask_t/vsf_i2c_status_t.
62 * To enable reimplementation, please enable macro below:
63 * VSF_I2C_CFG_REIMPLEMENT_TYPE_MODE for vsf_i2c_mode_t
64 * VSF_I2C_CFG_REIMPLEMENT_TYPE_CMD for vsf_i2c_cmd_t
65 * VSF_I2C_CFG_REIMPLEMENT_TYPE_STATUS for vsf_i2c_status_t
66 * VSF_I2C_CFG_REIMPLEMENT_TYPE_IRQ_MASK for vsf_i2c_irq_mask_t
67 * Reimplementation is used for optimization hw drivers, reimplement the bit mask according to hw registers.
68 * DO NOT reimplement these in non-hw drivers.
69 */
70
