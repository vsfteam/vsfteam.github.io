86typedef enum vsf_usart_mode_t {
87 // 0..1: STOP(13:12) in USART_CR2, shift right by 12
88 VSF_USART_0_5_STOPBIT = (1 << 12) >> 12,
89 VSF_USART_1_STOPBIT = (0 << 12) >> 12,
90 VSF_USART_1_5_STOPBIT = (3 << 12) >> 12,
91 VSF_USART_2_STOPBIT = (2 << 12) >> 12,
92
93 // 2..3: TE(3)/RE(2) in USART_CR1
94 VSF_USART_TX_ENABLE = (1 << 3),
95 VSF_USART_TX_DISABLE = (0 << 3),
96 VSF_USART_RX_ENABLE = (1 << 2),
97 VSF_USART_RX_DISABLE = (0 << 2),
98
99 // 4..5: CTSE(9)/RTSE(8) in USART_CR3, shift right by 4 to void conflict with PARITY
100 VSF_USART_NO_HWCONTROL = (0),
101 VSF_USART_RTS_HWCONTROL = (1 << 8) >> 4,
102 VSF_USART_CTS_HWCONTROL = (1 << 9) >> 4,
103 VSF_USART_RTS_CTS_HWCONTROL = VSF_USART_RTS_HWCONTROL
104 | VSF_USART_CTS_HWCONTROL,
105
106 // 6: HDSEL(3) in USART_CR3, shift left by 3
107 VSF_USART_HALF_DUPLEX_ENABLE = (1 << 3) << 3,
108 VSF_USART_HALF_DUPLEX_DISABLE = 0,
109
110 // 9..10: PCE(10)/PS(9) in USART_CR1
111 VSF_USART_NO_PARITY = (0 << 10),
112 VSF_USART_ODD_PARITY = (1 << 10) | (1 << 9),
113 VSF_USART_EVEN_PARITY = (1 << 10),
114
115 // 23: CLKEN(11) in USART_CR2, shift left by 12
116 VSF_USART_SYNC_CLOCK_ENABLE = (1 << 11) << 12,
117 VSF_USART_SYNC_CLOCK_DISABLE = (0 << 11) << 12,
118 // 22: CPOL(10) in USART_CR2, shift left by 12
119 VSF_USART_SYNC_CLOCK_POLARITY_LOW = (0 << 10) << 12,
120 VSF_USART_SYNC_CLOCK_POLARITY_HIGH = (1 << 10) << 12,
121 // 21: CPHA(9) in USART_CR2, shift left by 12
122 VSF_USART_SYNC_CLOCK_PHASE_1_EDGE = (0 << 9) << 12,
123 VSF_USART_SYNC_CLOCK_PHASE_2_EDGE = (1 << 9) << 12,
124 // 20: LBCL(8) in USART_CR2, shift left by 12
125 VSF_USART_SYNC_CLOCK_LAST_BIT_ENABLE = (1 << 8) << 12,
126 VSF_USART_SYNC_CLOCK_LAST_BIT_DISABLE = (0 << 8) << 12,
127#define VSF_USART_SYNC_CLOCK_LAST_BIT_ENABLE VSF_USART_SYNC_CLOCK_LAST_BIT_ENABLE
128#define VSF_USART_SYNC_CLOCK_LAST_BIT_DISABLE VSF_USART_SYNC_CLOCK_LAST_BIT_DISABLE
129#define VSF_USART_SYNC_CLOCK_LAST_BIT_MASK VSF_USART_SYNC_CLOCK_LAST_BIT_ENABLE | VSF_USART_SYNC_CLOCK_LAST_BIT_DISABLE
130
131 // 27..28: M(27..28) in USART_CR1
132 VSF_USART_6_BIT_LENGTH = (0 << 27),
133 VSF_USART_7_BIT_LENGTH = (1 << 27),
134 VSF_USART_8_BIT_LENGTH = (2 << 27),
135 VSF_USART_9_BIT_LENGTH = (3 << 27),
136
137 // more vendor specified modes can be added here
138
139 // 13: CMIE(13) in USART_CR1
140 VSF_USART_MUTE = (1 << 13),
141 // 15: SWAP(15) in USART_CR2
142 VSF_USART_SWAP = (1 << 15),
143 // 16..17: TXINV(17)/RXINV(16) in USART_CR2
144 VSF_USART_TX_INV = (1 << 17),
145 VSF_USART_RX_INV = (1 << 16),
146 // 14: OVER8(14) in USART_CR1
147 VSF_USART_OVERSAMPLE_8 = (1 << 14),
148 VSF_USART_OVERSAMPLE_16 = (0 << 14),
149 VSF_USART_OVERSAMPLE_MASK = (1 << 14),
150
151
152 __VSF_HW_USART_CR1_MASK = VSF_USART_6_BIT_LENGTH
153 | VSF_USART_9_BIT_LENGTH
154 | VSF_USART_8_BIT_LENGTH
155 | VSF_USART_7_BIT_LENGTH
156 | VSF_USART_NO_PARITY
157 | VSF_USART_ODD_PARITY
158 | VSF_USART_EVEN_PARITY
159 | VSF_USART_TX_ENABLE
160 | VSF_USART_RX_ENABLE
161 | VSF_USART_OVERSAMPLE_MASK
162 | VSF_USART_MUTE,
163 __VSF_HW_USART_CR2_MASK = VSF_USART_SYNC_CLOCK_ENABLE
164 | VSF_USART_SYNC_CLOCK_DISABLE
165 | VSF_USART_SYNC_CLOCK_POLARITY_LOW
166 | VSF_USART_SYNC_CLOCK_POLARITY_HIGH
167 | VSF_USART_SYNC_CLOCK_PHASE_1_EDGE
168 | VSF_USART_SYNC_CLOCK_PHASE_2_EDGE
169 | VSF_USART_SYNC_CLOCK_LAST_BIT_ENABLE
170 | VSF_USART_SYNC_CLOCK_LAST_BIT_DISABLE
171 | VSF_USART_0_5_STOPBIT
172 | VSF_USART_1_STOPBIT
173 | VSF_USART_1_5_STOPBIT
174 | VSF_USART_2_STOPBIT
175 | VSF_USART_TX_INV

177 | VSF_USART_SWAP,

179 | VSF_USART_HALF_DUPLEX_ENABLE,
180
181 // not supported, allocate unused bits
182 // 6..8
183 VSF_USART_10_BIT_LENGTH = (1 << 6),
184 VSF_USART_5_BIT_LENGTH = (2 << 6),
185 VSF_USART_FORCE_0_PARITY = (0 << 8),
186 VSF_USART_FORCE_1_PARITY = (1 << 8),
187
188 VSF_USART_TX_FIFO_THRESHOLD_NOT_FULL

198 VSF_USART_RX_FIFO_THRESHOLD_FULL = (2 << 22),

201 | VSF_USART_5_BIT_LENGTH
202 | VSF_USART_FORCE_0_PARITY
203 | VSF_USART_FORCE_1_PARITY,
204} vsf_usart_mode_t;

264typedef struct vsf_usart_status_t {
265 union {
266 struct {
267 uint32_t parity_err : 1; // PERR(0) in USART_STAT

270 uint32_t overrun_err : 1; // ORERR(3) in USART_STAT
271 uint32_t idle : 1; // IDLEF(4) in USART_STAT
272 uint32_t rfne : 1; // RFNE(5) in USART_STAT
273 uint32_t trans_complete : 1; // TC(6) in USART_STAT
274 uint32_t tfnf : 1; // TFNF(7) in USART_STAT
