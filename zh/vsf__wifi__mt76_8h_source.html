70/* MT76 bus operations.
71 *
72 * MT76 is a register-based chip (extensive cfg_read/cfg_write) plus chip-
73 * specific primitives for FCE writes, device commands, MCU command frames,
74 * TX frame submission and RX URB arming. The first member MUST be
75 * vsf_wifi_reg_bus_t so the generic wifi layer can treat the structure as a
76 * standard register bus; chip-private operations follow the shared part.
77 *
78 * On USB the bus driver maps reg_read/reg_write to ep0 vendor requests and
79 * the private ops to bulk URBs / ep0 control transfers; on other buses (e.g.
80 * SDIO) the same chip driver would use a different bus_ops implementation
81 * that maps to the appropriate host/controller primitives. */
82typedef struct vsf_wifi_mt76_bus_ops_t {
83 /* Generic register-bus interface. reg_read/reg_write are used for the
84 * many 32-bit configuration/register accesses that MT76 performs. */
85 vsf_wifi_reg_bus_t base;
86
87 /* Write an FCE (Forward Control Engine) DMA register. Used during
88 * firmware upload to set the destination address/length before sending
89 * the firmware payload as an MCU command frame. */
90 vsf_err_t (*fce_write)(vsf_wifi_t *wifi,
91 uint32_t addr, uint32_t val, vsf_wifi_done_t done);
92
93 /* Issue a device-mode control command (reset, IVB load).
94 * On USB this is an ep0 vendor request. */
95 vsf_err_t (*dev_cmd)(vsf_wifi_t *wifi,
96 uint8_t req, uint16_t value, uint16_t index, vsf_wifi_done_t done);
97
98 /* Issue a device-mode class request (patch enable / WMT reset).
99 * On USB this is an ep0 class request with a small data stage. */
100 vsf_err_t (*dev_class_cmd)(vsf_wifi_t *wifi,
101 uint8_t req, uint16_t value, uint16_t index,
102 const uint8_t *data, uint16_t len, vsf_wifi_done_t done);
103
104 /* Send an MCU command frame. On USB this is a bulk OUT transfer on the
105 * INBAND_CMD endpoint. The buffer already contains the TXINFO + CMD
106 * header + payload layout required by the chip. Completion is signalled
107 * when the bulk OUT URB finishes (the caller separately waits for the
108 * CMD_RESP via rx_submit). */
109 vsf_err_t (*mcu_cmd)(vsf_wifi_t *wifi,
110 const uint8_t *data, uint16_t len, vsf_wifi_done_t done);
111
112 /* Submit a TX 802.11 frame. On USB this is a bulk OUT transfer on one of
113 * the AC endpoints. The buffer already contains TXINFO + TXWI + frame. */
114 vsf_err_t (*tx_frame)(vsf_wifi_t *wifi,
115 const uint8_t *data, uint16_t len, uint8_t queue_idx, vsf_wifi_done_t done);
116
117 /* Start / re-arm a single RX reception. On USB this is a bulk IN URB. */
118 vsf_err_t (*rx_submit)(vsf_wifi_t *wifi,
119 uint8_t *buf, uint16_t len, uint8_t queue_idx);
120} vsf_wifi_mt76_bus_ops_t;
