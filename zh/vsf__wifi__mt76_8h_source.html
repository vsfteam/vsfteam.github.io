43/* MT76 bus operations.
44 *
45 * These are chip-level communication primitives, not USB-specific calls. On
46 * USB the bus driver maps them to ep0 vendor requests and bulk URBs; on other
47 * buses (e.g. SDIO) the same chip driver would use a different bus_ops
48 * implementation that maps to the appropriate host/controller primitives. */
49typedef struct vsf_wifi_mt76_bus_ops_t {
50 /* Called by the chip driver once initialization is complete and the data
51 * path can be enabled. The bus driver starts RX URBs / DMA here. */
52 void (*on_ready)(vsf_wifi_t *wifi);
53
54 /* Read a 32-bit configuration/register word. On USB this becomes an
55 * ep0 vendor IN request; the exact layout is bus-driver private. */
56 vsf_err_t (*cfg_read)(vsf_wifi_t *wifi,
57 uint32_t addr, uint32_t *out, vsf_wifi_done_t done);
58
59 /* Write a 32-bit configuration/register word. On USB this becomes an
60 * ep0 vendor OUT request; the exact layout is bus-driver private. */
61 vsf_err_t (*cfg_write)(vsf_wifi_t *wifi,
62 uint32_t addr, uint32_t val, vsf_wifi_done_t done);
63
64 /* Write an FCE (Forward Control Engine) DMA register. Used during
65 * firmware upload to set the destination address/length before sending
66 * the firmware payload as an MCU command frame. */
67 vsf_err_t (*fce_write)(vsf_wifi_t *wifi,
68 uint32_t addr, uint32_t val, vsf_wifi_done_t done);
69
70 /* Issue a device-mode control command (reset, IVB load).
71 * On USB this is an ep0 vendor request. */
72 vsf_err_t (*dev_cmd)(vsf_wifi_t *wifi,
73 uint8_t req, uint16_t value, uint16_t index, vsf_wifi_done_t done);
74
75 /* Issue a device-mode class request (patch enable / WMT reset).
76 * On USB this is an ep0 class request with a small data stage. */
77 vsf_err_t (*dev_class_cmd)(vsf_wifi_t *wifi,
78 uint8_t req, uint16_t value, uint16_t index,
79 const uint8_t *data, uint16_t len, vsf_wifi_done_t done);
80
81 /* Send an MCU command frame. On USB this is a bulk OUT transfer on the
82 * INBAND_CMD endpoint. The buffer already contains the TXINFO + CMD
83 * header + payload layout required by the chip. Completion is signalled
84 * when the bulk OUT URB finishes (the caller separately waits for the
85 * CMD_RESP via rx_submit). */
86 vsf_err_t (*mcu_cmd)(vsf_wifi_t *wifi,
87 const uint8_t *data, uint16_t len, vsf_wifi_done_t done);
88
89 /* Submit a TX 802.11 frame. On USB this is a bulk OUT transfer on one of
90 * the AC endpoints. The buffer already contains TXINFO + TXWI + frame. */
91 vsf_err_t (*tx_frame)(vsf_wifi_t *wifi,
92 const uint8_t *data, uint16_t len, uint8_t queue_idx, vsf_wifi_done_t done);
93
94 /* Start / re-arm a single RX reception. On USB this is a bulk IN URB. */
95 vsf_err_t (*rx_submit)(vsf_wifi_t *wifi,
96 uint8_t *buf, uint16_t len, uint8_t queue_idx);
97} vsf_wifi_mt76_bus_ops_t;
