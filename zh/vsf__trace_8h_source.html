44// display flag
45#define VSF_TRACE_DF_DS(n) (((n) & 0xFF) << 0) // data size
46#define VSF_TRACE_DF_DPL(n) (((n) & 0xFF) << 8) // data per line
47#define VSF_TRACE_DF_ADDR (1UL << 16) // display address
48#define VSF_TRACE_DF_CHAR (1UL << 17) // display character
49#define VSF_TRACE_DF_NEWLINE (1UL << 18) // append newline
50
51#define VSF_TRACE_DF_U8_16 (VSF_TRACE_DF_DS(1) | VSF_TRACE_DF_DPL(16))
52#define VSF_TRACE_DF_U8_16_A (VSF_TRACE_DF_U8_16 | VSF_TRACE_DF_ADDR)
53#define VSF_TRACE_DF_U8_16_AC (VSF_TRACE_DF_U8_16_A | VSF_TRACE_DF_CHAR)
54#define VSF_TRACE_DF_U8_16_ACN (VSF_TRACE_DF_U8_16_AC | VSF_TRACE_DF_NEWLINE)
55#define VSF_TRACE_DF_U8_16_AN (VSF_TRACE_DF_U8_16_A | VSF_TRACE_DF_NEWLINE)
56#define VSF_TRACE_DF_U8_16_N (VSF_TRACE_DF_U8_16 | VSF_TRACE_DF_NEWLINE)
