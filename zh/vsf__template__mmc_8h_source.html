255/* SD/MMC version bits; 8 flags, 8 major, 8 minor, 8 change */
256#define SD_VERSION_SD (1U << 31)
257#define MMC_VERSION_MMC (1U << 30)
258
259#define MAKE_SDMMC_VERSION(a, b, c) \
260 ((((uint32_t)(a)) << 16) | ((uint32_t)(b) << 8) | (uint32_t)(c))
261#define MAKE_SD_VERSION(a, b, c) \
262 (SD_VERSION_SD | MAKE_SDMMC_VERSION(a, b, c))
263#define MAKE_MMC_VERSION(a, b, c) \
264 (MMC_VERSION_MMC | MAKE_SDMMC_VERSION(a, b, c))
265
266#define EXTRACT_SDMMC_MAJOR_VERSION(x) \
267 (((uint32_t)(x) >> 16) & 0xff)
268#define EXTRACT_SDMMC_MINOR_VERSION(x) \
269 (((uint32_t)(x) >> 8) & 0xff)
270#define EXTRACT_SDMMC_CHANGE_VERSION(x) \
271 ((uint32_t)(x) & 0xff)
