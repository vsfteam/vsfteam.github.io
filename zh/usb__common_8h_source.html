301static inline int vsf_usb_hex_to_bin(char ch)
302{
303 if ((ch >= '0') && (ch <= '9')) {
304 return ch - '0';
305 }
306 ch = tolower(ch);
307 if ((ch >= 'a') && (ch <= 'f')) {
308 return ch - 'a' + 10;
309 }
310 return -1;
311}
