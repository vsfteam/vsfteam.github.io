147// reg: PIXEL_FORAMT high 4 bits
148#define MIPI_PIXEL_FORMAT_DPI_3BITS (1 << 4)
149#define MIPI_PIXEL_FORMAT_DPI_8BITS (2 << 4)
150#define MIPI_PIXEL_FORMAT_DPI_12BITS (3 << 4)
151#define MIPI_PIXEL_FORMAT_DPI_16BITS (5 << 4)
152#define MIPI_PIXEL_FORMAT_DPI_18BITS (6 << 4)
153#define MIPI_PIXEL_FORMAT_DPI_24BITS (7 << 4)
154// reg: PIXEL_FORAMT low 4 bits
155#define MIPI_PIXEL_FORMAT_DBI_3BITS (1)
156#define MIPI_PIXEL_FORMAT_DBI_8BITS (2)
157#define MIPI_PIXEL_FORMAT_DBI_12BITS (3)
158#define MIPI_PIXEL_FORMAT_DBI_16BITS (5)
159#define MIPI_PIXEL_FORMAT_DBI_18BITS (6)
160#define MIPI_PIXEL_FORMAT_DBI_24BITS (7)
161
162#ifndef MIPI_LCD_PIXEL_FORMAT_VALUE
163# define MIPI_LCD_PIXEL_FORMAT_VALUE (MIPI_PIXEL_FORMAT_DPI_16BITS | MIPI_PIXEL_FORMAT_DBI_16BITS)
164#endif

166// These 3 bits control the direction from the MPU to memory write/read.
167#define GRAM_WRITE_TOP_TO_BOTTOM (0 << 7)
168#define GRAM_WRITE_BOTTOM_TO_TOP (1 << 7)
169#define GRAM_WRITE_LEFT_TO_RIGHT (0 << 6)
170#define GRAM_WRITE_RIGHT_TO_LEFT (1 << 6)
171#define GRAM_WRITE_NORMAL_MODE (0 << 5)
172#define GRAM_WRITE_ROW_COLUMN_EXCHANGE (1 << 5)
173
174// LCD vertical refresh direction control.
175#define LCD_REFRESH_TOP_TO_BOTTOM (0 << 4)
176#define LCD_REFRESH_BOTTOM_TO_TOP (1 << 4)
177#define LCD_REFRESH_RGB_ORDER (0 << 3)
178#define LCD_REFRESH_BGR_ORDER (1 << 3)
179#define LCD_REFRESH_LEFT_TO_RIGHT (0 << 2)
180#define LCD_REFRESH_RIGHT_TO_LEFT (1 << 2)
181#ifndef MIPI_LCD_ADDRESS_MODE_VALUE
182# define MIPI_LCD_ADDRESS_MODE_VALUE (GRAM_WRITE_TOP_TO_BOTTOM | GRAM_WRITE_LEFT_TO_RIGHT | GRAM_WRITE_NORMAL_MODE)
183#endif
