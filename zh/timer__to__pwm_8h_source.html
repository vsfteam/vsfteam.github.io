59/*
60 * Generic hw PWM instance that maps every call onto a vsf_timer_t.
61 *
62 * .timer: pointer to the underlying timer class instance (cast from the
63 * concrete vsf_hw_timer_t* on init-time). Uses vsf_timer_t so the
64 * glue stays chip-agnostic and goes through the dispatcher.
65 * REQUIRES VSF_TIMER_CFG_MULTI_CLASS == ENABLED.
66 * .channel: timer channel bound to this PWM instance (0 for most chips).
67 * .cfg: cached user config (for get_configuration).
68 * .period: cached full period in timer ticks, written by pwm_set(), used by
69 * pwm_enable() to re-arm after a stop/enable cycle.
70 */
71typedef struct vsf_hw_pwm_t {
72#if VSF_HW_PWM_CFG_MULTI_CLASS == ENABLED
73 vsf_pwm_t vsf_pwm;
74#endif
75 vsf_timer_t *timer;
76 uint8_t channel;
77 bool enabled;
78 bool configured;
79 vsf_pwm_cfg_t cfg;
80 uint32_t period;
81 uint32_t pulse;
82} vsf_hw_pwm_t;
