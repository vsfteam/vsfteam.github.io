330 return (vsf_spi_mode_t)((bits - VSF_SPI_DATASIZE_DIFF) << VSF_SPI_DATASIZE_BIT_OFFSET);
331}
332#endif

334#if VSF_SPI_CFG_REIMPLEMENT_MODE_TO_BYTES == ENABLD
335static inline uint8_t vsf_spi_mode_to_data_bytes(vsf_spi_mode_t mode)
336{
337 int bits = (mode & VSF_SPI_DATASIZE_MASK);
338 if (bits <= VSF_SPI_DATASIZE_8) {
339 return 1;
340 } else if (bits <= VSF_SPI_DATASIZE_16) {
341 return 2;
342 } else {
343 return 4;
344 }
