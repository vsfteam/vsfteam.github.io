249#define VSF_USART_ADDITIONAL_APIS(__prefix_name) \
250 static inline vsf_err_t VSF_MCONNECT(__prefix_name, _send_break) \
251 (VSF_MCONNECT(__prefix_name, _t) *usart_ptr) \
252 { \
253 return VSF_MCONNECT(__prefix_name, _ctrl)(usart_ptr, \
254 VSF_USART_CTRL_SEND_BREAK, NULL); \
255 } \
256 static inline vsf_err_t VSF_MCONNECT(__prefix_name, _set_break) \
257 (VSF_MCONNECT(__prefix_name, _t) *usart_ptr) \
258 { \
259 vsf_usart_capability_t cap = \
260 VSF_MCONNECT(__prefix_name, _capability)(usart_ptr); \
261 VSF_HAL_ASSERT(cap.support_set_and_clear_break); \
262 return VSF_MCONNECT(__prefix_name, _ctrl)(usart_ptr, \
263 VSF_USART_CTRL_SET_BREAK, NULL); \
264 } \
265 static inline vsf_err_t VSF_MCONNECT(__prefix_name, _clear_break) \
266 (VSF_MCONNECT(__prefix_name, _t) *usart_ptr) \
267 { \
268 vsf_usart_capability_t cap = \
269 VSF_MCONNECT(__prefix_name, _capability)(usart_ptr); \
270 VSF_HAL_ASSERT(cap.support_set_and_clear_break); \
271 return VSF_MCONNECT(__prefix_name, _ctrl)(usart_ptr, \
272 VSF_USART_CTRL_CLEAR_BREAK, NULL); \
273 }
