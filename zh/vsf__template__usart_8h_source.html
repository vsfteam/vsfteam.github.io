583static inline uint8_t vsf_usart_mode_to_data_bits(vsf_usart_mode_t mode)
584{
585 switch (mode & VSF_USART_BIT_LENGTH_MASK) {
586 case VSF_USART_5_BIT_LENGTH:
587 return 5;
588 case VSF_USART_6_BIT_LENGTH:
589 return 6;
590 case VSF_USART_7_BIT_LENGTH:
591 return 7;
592 case VSF_USART_8_BIT_LENGTH:
593 return 8;
594 case VSF_USART_9_BIT_LENGTH:
595 return 9;
596 default:
597 return 0;
598 }
599}
