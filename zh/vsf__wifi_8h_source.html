128/*
129 * One unit of register-access work — write `val` (LE32) to register `reg`.

131 * BBP / RF indirect writes are NOT a separate op type: chip backends pre-

133 * This keeps the wifi state machine free of chip-specific knowledge.
134 */
135typedef struct vsf_wifi_op_t {

137 uint32_t val;

140/*
141 * Bulk register-block payload (firmware blob, calibration table, etc.).
142 * Each chunk is shipped via bus_ops->reg_block_write (or, when the bus
143 * lacks that primitive, decomposed into multiple reg_write calls by the
144 * generic layer). `data` MUST stay valid until the completion callback
145 * fires — pointing at .rodata is fine on every supported bus.
146 */
147typedef struct vsf_wifi_blob_t {
148 const uint8_t *data;
149 uint32_t len;
150 uint16_t base_reg;
151 uint16_t chunk_size; /* hint; bus impl may override */
152} vsf_wifi_blob_t;

154/*============================ BUS OPS =======================================*
155 *
156 * Only the *most stable* bus primitives live in this vtable: single-register
157 * 32-bit write / read, plus an optional contiguous register-block write.
158 * These are guaranteed to be expressible on any addressable wifi NIC bus
159 * (USB ep0 vendor request, SDIO CMD52/CMD53, SPI register frame, MMIO).
160 *
161 * Higher-level concepts that vary across buses (bulk endpoint TX/RX, port
162 * reset, PHY interrupts, doorbells) are intentionally NOT abstracted here
163 * — those continue to live inside the bus driver until enough chip / bus
164 * combinations exist to draw a clean line.
165 *
166 * Concurrency contract: at most one bus_ops call may be in flight at any
167 * time. Issuing a second call before the previous `done` callback has
168 * fired is a programming error; the bus driver MAY return VSF_ERR_NOT_
169 * AVAILABLE in that case.
170 *==========================================================================*/
171
172struct vsf_wifi_bus_ops_t {
173 /* Single-register 32-bit write. REQUIRED.
174 *
175 * Returns VSF_ERR_NONE if the write was queued; completion is delivered
176 * later via `done`. `done` may be NULL for fire-and-forget. Synchronous
177 * errors (resource exhaustion, bus busy) are returned directly without
178 * invoking `done`.
179 */
180 vsf_err_t (*reg_write) (vsf_wifi_t *wifi, uint16_t reg, uint32_t val,
181 vsf_wifi_done_t done);
182
183 /* Single-register 32-bit read. REQUIRED.
184 *
185 * `out` is written before `done` fires; caller must keep it valid until
186 * then. Same async / error semantics as reg_write.
187 */
188 vsf_err_t (*reg_read) (vsf_wifi_t *wifi, uint16_t reg, uint32_t *out,
189 vsf_wifi_done_t done);
190
191 /* Contiguous register-block write (firmware upload / calibration table
192 * / TX descriptor RAM). OPTIONAL — when NULL the generic layer falls
193 * back to a sequence of reg_write calls.
194 *
195 * `base` is the starting register/memory address; the bus implementation
196 * is free to fragment the payload internally (USB ep0 picks 64-byte
197 * vendor-request chunks, SDIO CMD53 picks block size, etc.). `data`
198 * must stay valid until `done` fires.
199 */
200 vsf_err_t (*reg_block_write)(vsf_wifi_t *wifi, uint16_t base,
201 const uint8_t *data, uint32_t len,
202 vsf_wifi_done_t done);
203
204 /* Raw ep0 vendor request with no data stage (wLength=0). OPTIONAL —
205 * NULL when the bus has no such concept. Used by USB chips to issue
206 * USB_DEVICE_MODE (start MCU firmware / reset the digital core), which
207 * is NOT a register write. `request`=bRequest, `value`=wValue,
208 * `index`=wIndex. Same async / error semantics as reg_write. */
209 vsf_err_t (*vendor_request) (vsf_wifi_t *wifi, uint8_t request,
210 uint16_t value, uint16_t index,
211 vsf_wifi_done_t done);
212
213 /* Bus-layer ready notification. OPTIONAL — called when the wifi init
214 * chain completes successfully (before vsf_wifi_on_ready). The bus
215 * driver uses this to enable the data path (e.g. submit bulk RX URBs).
216 */
217 void (*on_ready)(vsf_wifi_t *wifi);
218
219 /* Bulk data-frame transmit. OPTIONAL — NULL on buses that have no bulk
220 * data endpoint (pure ep0). `data` points at a fully chip-encoded TX
221 * payload (built by drv->build_tx); `len` is its total length. Unlike
222 * the reg/script primitives this is fire-and-forget: the bus driver
223 * owns completion internally (TX URB pool). Returns VSF_ERR_NONE when
224 * the frame was queued. */
225 vsf_err_t (*data_tx)(vsf_wifi_t *wifi, uint8_t *data, uint16_t len);
226};

228/*============================ CHIP DRIVER VTABLE ============================*
229 *
230 * All chip operations are async: the chip driver builds an op-script (static
231 * .rodata or wifi->scratch_ops) and submits it via vsf_wifi_run_script. The
232 * outer `done` callback fires when the entire script has been ACKed by the
233 * device. Returning VSF_ERR_NONE from these hooks means the script was
234 * successfully queued; actual completion arrives later via `done`.
235 *
236 * The wifi layer never calls bus_ops directly — it always goes through this
237 * chip vtable. Conversely the chip code only sees bus_ops, never the bus
238 * driver's private types.
239 *==========================================================================*/
240
241struct vsf_wifi_chip_drv_t {
242 const char *name;
243
244 /* Optional firmware uploader (rt2870.bin etc.). Invoked once before
245 * init() during attach. Pure ops can use vsf_wifi_run_script; bulk
246 * blob uploads call vsf_wifi_run_blob. Leave NULL when the chip needs
247 * no firmware. */
248 vsf_err_t (*firmware_load)(vsf_wifi_t *wifi, vsf_wifi_done_t done);
249 vsf_err_t (*init) (vsf_wifi_t *wifi, vsf_wifi_done_t done);
250 void (*fini) (vsf_wifi_t *wifi);
251 vsf_err_t (*set_channel) (vsf_wifi_t *wifi, uint8_t channel,
252 vsf_wifi_done_t done);
253 vsf_err_t (*set_rx_filter)(vsf_wifi_t *wifi, uint32_t mask,
254 vsf_wifi_done_t done);
255 vsf_err_t (*set_mac_addr) (vsf_wifi_t *wifi, const uint8_t mac[6],
256 vsf_wifi_done_t done);
257 vsf_err_t (*set_bssid) (vsf_wifi_t *wifi, const uint8_t bssid[6],
258 vsf_wifi_done_t done);
259 vsf_err_t (*set_auth_mode)(vsf_wifi_t *wifi,
260 const vsf_wifi_auth_cfg_t *cfg,
261 vsf_wifi_done_t done);
262 vsf_err_t (*connect) (vsf_wifi_t *wifi,
263 const uint8_t bssid[6], const uint8_t *ssid,
264 uint8_t ssid_len, uint8_t channel,
265 vsf_wifi_done_t done);
266 vsf_err_t (*disconnect) (vsf_wifi_t *wifi, vsf_wifi_done_t done);
267 vsf_err_t (*get_link_info)(vsf_wifi_t *wifi,
268 vsf_wifi_link_info_t *info); /* sync */
269
270 /* Optional RX-frame parser used during scan. When non-NULL and the
271 * driver is currently scanning, the wifi layer routes incoming frames
272 * here instead of vsf_wifi_on_rx; the chip code extracts beacons /
273 * probe responses and emits vsf_wifi_on_scan_result(). */
274 void (*parse_rx) (vsf_wifi_t *wifi, uint8_t *frame, uint16_t len);
275
276 /* Optional TX-descriptor builder. Wraps a raw 802.11 frame into the
277 * chip-specific on-wire TX layout (e.g. RT2800 TXINFO + TXWI + frame +
278 * pad) by writing into `dst` (capacity `dst_cap`). Returns the total
279 * byte count to ship over the data bus, or 0 on failure (dst too small
280 * / unsupported). Bus-agnostic: the chip only knows the descriptor
281 * format; the bus driver (bus_ops->data_tx) performs the actual send. */
282 uint16_t (*build_tx) (vsf_wifi_t *wifi, uint8_t *dst, uint16_t dst_cap,
283 const uint8_t *frame, uint16_t frame_len);
284};

364/*
365 * One-shot construction. After this call:
366 * - wifi->drv / wifi->bus_ops are bound;
367 * - all state fields are zero / cleared;

369 * - the wifi is NOT started yet (call vsf_wifi_start when ready).

422/*
423 * Run a sequence of register writes asynchronously.
424 *
425 * ops : array of (reg, val) pairs. May point to .rodata or to
426 * wifi->scratch_ops. Caller MUST keep `ops` valid until
427 * `done` fires.
428 * count : number of ops to execute (must be > 0 for a real submission).
429 * done : completion callback (NULL = fire-and-forget).
430 *
431 * Returns VSF_ERR_NONE on accepted submission (completion via `done`),
432 * VSF_ERR_NOT_AVAILABLE when another script / blob is in flight, or a bus-
433 * level synchronous error otherwise.
434 */
435vsf_err_t vsf_wifi_run_script(vsf_wifi_t *wifi,
436 const vsf_wifi_op_t *ops, uint16_t count,
437 vsf_wifi_done_t done);
438
439/*
440 * Stream a contiguous register-block payload. Internally calls
441 * bus_ops->reg_block_write when available; otherwise decomposes into
442 * ceil(len / 4) reg_write calls.
443 *
444 * Same concurrency rules as vsf_wifi_run_script: at most one script /
445 * blob in flight per wifi.
446 */
447vsf_err_t vsf_wifi_run_blob(vsf_wifi_t *wifi,
448 const vsf_wifi_blob_t *blob,
449 vsf_wifi_done_t done);
450
451/*
452 * Issue a single ep0 vendor request with no data stage (wLength=0) via
453 * bus_ops->vendor_request. Used by USB chips for control commands that are
454 * not register writes (e.g. RT2800 USB_DEVICE_MODE to start the MCU firmware
455 * or reset the digital core). Completes with VSF_ERR_NOT_SUPPORTED when the
456 * bus has no vendor_request primitive.
457 *
458 * Same concurrency rules as vsf_wifi_run_script: at most one script / blob /
459 * vendor request in flight per wifi.
460 */
461vsf_err_t vsf_wifi_run_vendor(vsf_wifi_t *wifi, uint8_t request,
462 uint16_t value, uint16_t index,
463 vsf_wifi_done_t done);
464
465/*
466 * Periodically read `reg` until `match(val)` returns true.
467 *
468 * reg : MAC register to poll.
469 * match : predicate; the read value is fed in on every iteration.
470 * max_retry : upper bound on read attempts (>= 1). The first read
471 * counts as attempt #1.
472 * interval_ms : delay between consecutive reads. 0 means "re-issue
473 * immediately on completion" (use only if the bus already
474 * paces reads, e.g. USB ep0 vendor request).
475 * done : completion callback.
476 * VSF_ERR_NONE : predicate matched.
477 * VSF_ERR_TIMEOUT : retries exhausted without match.
478 * other : bus-level error.
479 *
480 * Same single-flight constraint as run_script / run_blob.
481 */
482vsf_err_t vsf_wifi_run_read_poll(vsf_wifi_t *wifi, uint16_t reg,
483 vsf_wifi_match_fn_t match,
484 uint16_t max_retry, uint16_t interval_ms,
485 vsf_wifi_done_t done);
486
487/*
488 * Single 32-bit register read. The result is stored at *out before
489 * `done` fires; caller must keep `out` valid until then. Same single-
490 * flight constraint as run_script / run_blob / run_read_poll.
491 */
492vsf_err_t vsf_wifi_run_read(vsf_wifi_t *wifi, uint16_t reg, uint32_t *out,
493 vsf_wifi_done_t done);
