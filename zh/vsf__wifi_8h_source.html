165/*
166 * One unit of register-access work — write `val` (LE32) to register `reg`.
167 *
168 * BBP / RF indirect writes are NOT a separate op type: chip backends pre-
169 * encode them (see RT_OP_BBP / RT_OP_RF in chip/rt28xx/vsf_wifi_rt28xx.c).
170 * This keeps the wifi state machine free of chip-specific knowledge.
171 */
172typedef struct vsf_wifi_reg_op_t {
173 uint16_t reg;
174 uint32_t val;
175} vsf_wifi_reg_op_t;

177/*
178 * Bulk register-block payload (firmware blob, calibration table, etc.).
179 * Each chunk is shipped via reg_bus->reg_block_write (or, when the bus
180 * lacks that primitive, decomposed into multiple reg_write calls by the
181 * generic layer). `data` MUST stay valid until the completion callback
182 * fires — pointing at .rodata is fine on every supported bus.
183 */
184typedef struct vsf_wifi_reg_blob_t {
185 const uint8_t *data;
186 uint32_t len;
187 uint16_t base_reg;
188 uint16_t chunk_size; /* hint; bus impl may override */
189} vsf_wifi_reg_blob_t;

191/*============================ REGISTER BUS OPS ==============================*
192 *
193 * Optional register-access helper for register-based WiFi chips (e.g.
194 * Ralink RT2X00). Chips that do NOT speak a register protocol — such as
195 * MediaTek mt76, which uses firmware commands over bulk/interrupt endpoints
196 * — can leave wifi->reg_bus NULL and implement their own transport inside
197 * drv->tx / drv->init / etc.
198 *
199 * Concurrency contract: at most one reg_bus call may be in flight at any
200 * time. Issuing a second call before the previous `done` callback has
201 * fired is a programming error; the bus driver MAY return VSF_ERR_NOT_
202 * AVAILABLE in that case.
203 *==========================================================================*/
204
205struct vsf_wifi_reg_bus_t {
206 /* Single-register 32-bit write. REQUIRED.
207 *
208 * Returns VSF_ERR_NONE if the write was queued; completion is delivered
209 * later via `done`. `done` may be NULL for fire-and-forget. Synchronous
210 * errors (resource exhaustion, bus busy) are returned directly without
211 * invoking `done`.
212 */
213 vsf_err_t (*reg_write) (vsf_wifi_t *wifi, uint16_t reg, uint32_t val,
214 vsf_wifi_done_t done);
215
216 /* Single-register 32-bit read. REQUIRED.
217 *
218 * `out` is written before `done` fires; caller must keep it valid until
219 * then. Same async / error semantics as reg_write.
220 */
221 vsf_err_t (*reg_read) (vsf_wifi_t *wifi, uint16_t reg, uint32_t *out,
222 vsf_wifi_done_t done);
223
224 /* Contiguous register-block write (firmware upload / calibration table
225 * / TX descriptor RAM). OPTIONAL — when NULL the generic layer falls
226 * back to a sequence of reg_write calls.
227 *
228 * `base` is the starting register/memory address; the bus implementation
229 * is free to fragment the payload internally (USB ep0 picks 64-byte
230 * vendor-request chunks, SDIO CMD53 picks block size, etc.). `data`
231 * must stay valid until `done` fires.
232 */
233 vsf_err_t (*reg_block_write)(vsf_wifi_t *wifi, uint16_t base,
234 const uint8_t *data, uint32_t len,
235 vsf_wifi_done_t done);
236
237 /* Raw ep0 vendor request with no data stage (wLength=0). OPTIONAL —
238 * NULL when the bus has no such concept. Used by USB chips to issue
239 * USB_DEVICE_MODE (start MCU firmware / reset the digital core), which
240 * is NOT a register write. `request`=bRequest, `value`=wValue,
241 * `index`=wIndex. Same async / error semantics as reg_write. */
242 vsf_err_t (*vendor_request) (vsf_wifi_t *wifi, uint8_t request,
243 uint16_t value, uint16_t index,
244 vsf_wifi_done_t done);
245
246 /* Bus-layer ready notification. OPTIONAL — called when the wifi init
247 * chain completes successfully (before vsf_wifi_on_ready). The bus
248 * driver uses this to enable the data path (e.g. submit bulk RX URBs).
249 */
250 void (*on_ready)(vsf_wifi_t *wifi);
251
252 /* Bulk data-frame transmit for register-based buses that also have a
253 * bulk data endpoint. OPTIONAL — NULL when the register bus has no data
254 * endpoint (pure ep0) or when the chip drv->tx handles data transmission
255 * itself. `data` points at a fully chip-encoded TX payload; `len` is its
256 * total length. Fire-and-forget: the bus driver owns completion internally
257 * (TX URB pool). Returns VSF_ERR_NONE when the frame was queued. */
258 vsf_err_t (*data_tx)(vsf_wifi_t *wifi, uint8_t *data, uint16_t len);
259};

261/*============================ CHIP DRIVER VTABLE ============================*
262 *
263 * All chip operations are async: the chip driver builds an op-script (static
264 * .rodata or wifi->scratch_ops) and submits it via vsf_wifi_reg_run_script. The
265 * outer `done` callback fires when the entire register script has been ACKed by the
266 * device. Returning VSF_ERR_NONE from these hooks means the script was
267 * successfully queued; actual completion arrives later via `done`.
268 *
269 * The wifi layer never calls reg_bus directly — it always goes through the
270 * chip vtable. Register-based chips use the helpers vsf_wifi_reg_run_script /
271 * run_blob / reg_read / reg_read_poll; command-based chips ignore reg_bus.
272 *==========================================================================*/
273

275/*
276 * Optional hardware crypto backend. When a chip driver supplies crypto_ops
277 * with a non-NULL install_key, the wifi layer hands the negotiated PTK.TK /
278 * GTK to the hardware engine and assumes the chip performs CCMP in-line; the
279 * software CCMP path (wpa_hw_crypto == false) is then skipped on both TX and
280 * RX. Leaving crypto_ops (or install_key) NULL selects the built-in software
281 * CCMP fallback.
282 *
283 * install_key : program a key. key_idx 0 + pairwise == the unicast TK;
284 * key_idx 1..3 + !pairwise == a GTK. `mac` is the peer for
285 * pairwise keys, NULL for group keys.
286 * `done` is called when the register script finishes; the
287 * caller may be in an EDA or interrupt context, so
288 * install_key MUST NOT block. Returns VSF_ERR_NONE on
289 * successful submission.
290 * encrypt/decrypt : reserved per-frame overrides for chips that still need
291 * wifi-layer framing with a chip-specific tweak; NULL means
292 * "use the built-in software CCMP".
293 */
294typedef struct vsf_wifi_crypto_ops_t {
295 vsf_err_t (*install_key)(vsf_wifi_t *wifi, uint8_t key_idx, bool pairwise,
296 const uint8_t *key, uint8_t key_len,
297 const uint8_t *mac, vsf_wifi_done_t done);
298 vsf_err_t (*encrypt) (vsf_wifi_t *wifi, uint8_t *dot11,
299 uint16_t *len, uint16_t cap);
300 vsf_err_t (*decrypt) (vsf_wifi_t *wifi, uint8_t *dot11, uint16_t *len);
301} vsf_wifi_crypto_ops_t;
302#endif

384/*============================ NETIF BINDING ================================*
385 *
386 * Runtime binding for a network-stack adapter (e.g. the lwIP netdrv driver
387 * in vsf_wifi_netdrv.c). Once a backend is attached via
388 * vsf_wifi_netdrv_attach(), the core delivers decrypted RX data frames to
389 * its on_rx (replacing the weak vsf_wifi_on_rx hook), and additionally
390 * delivers link up/down events to its on_link_up / on_link_down. Link
391 * events still also fire the weak vsf_wifi_on_link_up / vsf_wifi_on_link_down
392 * hooks: those are control-plane notifications (e.g. unblocking a connect
393 * command) and run alongside the backend's data-plane handling. With no
394 * backend attached only the weak hooks are used (back-compat with the
395 * diagnostic commands). All callbacks run in the bus driver's EDA context,
396 * exactly like the weak hooks. `on_rx` receives a naked, decrypted 802.11
397 * data frame (FC .. payload); the backend is responsible for any 802.11 <->
398 * Ethernet framing. */
399typedef struct vsf_wifi_netif_ops_t {
400 void (*on_rx) (void *param, vsf_wifi_t *wifi,
401 uint8_t *frame, uint16_t len);
402 void (*on_link_up) (void *param, vsf_wifi_t *wifi,
403 const vsf_wifi_link_info_t *info);
404 void (*on_link_down)(void *param, vsf_wifi_t *wifi, uint8_t reason);
405} vsf_wifi_netif_ops_t;

472/*============================ REGISTER-BUS LAYER API =========================*
473 *
474 * These are exported so a bus driver can drive the wifi state machine.
475 * Application code does not call them directly.

478/*
479 * One-shot construction. After this call:
480 * - wifi->drv is bound;
481 * - wifi->reg_bus is bound (may be NULL for non-register-based chips);
482 * - all state fields are zero / cleared;
483 * - scan timer is initialised but not armed;
484 * - the wifi is NOT started yet (call vsf_wifi_start when ready).
485 *
486 * `post_eda` is the EDA on which the wifi layer will post follow-up events
487 * (scan-hop callbacks). Typically the bus driver passes its own EDA so
488 * that timer-driven hops are handled in the same context as bus completions.
489 */
490void vsf_wifi_init(vsf_wifi_t *wifi,
491 const vsf_wifi_chip_drv_t *drv,
492 const vsf_wifi_reg_bus_t *reg_bus,

577/*
578 * Run a sequence of register writes asynchronously.
579 *
580 * ops : array of (reg, val) pairs. May point to .rodata or to
581 * wifi->scratch_ops. Caller MUST keep `ops` valid until
582 * `done` fires.
583 * count : number of ops to execute (must be > 0 for a real submission).
584 * done : completion callback (NULL = fire-and-forget).
585 *
586 * Returns VSF_ERR_NONE on accepted submission (completion via `done`),
587 * VSF_ERR_NOT_AVAILABLE when another script / blob is in flight, or a bus-
588 * level synchronous error otherwise.
589 */
590vsf_err_t vsf_wifi_reg_run_script(vsf_wifi_t *wifi,
591 const vsf_wifi_reg_op_t *ops, uint16_t count,
592 vsf_wifi_done_t done);

594/*
595 * Stream a contiguous register-block payload. Internally calls
596 * reg_bus->reg_block_write when available; otherwise decomposes into
597 * ceil(len / 4) reg_write calls.
598 *
599 * Same concurrency rules as vsf_wifi_reg_run_script: at most one script /
600 * blob in flight per wifi.
601 */
602vsf_err_t vsf_wifi_reg_run_blob(vsf_wifi_t *wifi,
603 const vsf_wifi_reg_blob_t *blob,
604 vsf_wifi_done_t done);
605
606/*
607 * Issue a single ep0 vendor request with no data stage (wLength=0) via
608 * reg_bus->vendor_request. Used by USB chips for control commands that are
609 * not register writes (e.g. RT2800 USB_DEVICE_MODE to start the MCU firmware
610 * or reset the digital core). Completes with VSF_ERR_NOT_SUPPORTED when the
611 * bus has no vendor_request primitive.
612 *
613 * Same concurrency rules as vsf_wifi_reg_run_script: at most one script / blob /
614 * vendor request in flight per wifi.
615 */
616vsf_err_t vsf_wifi_reg_run_vendor(vsf_wifi_t *wifi, uint8_t request,
617 uint16_t value, uint16_t index,
618 vsf_wifi_done_t done);
619
620/*
621 * Periodically read `reg` until `match(val)` returns true.
622 *
623 * reg : MAC register to poll.
624 * match : predicate; the read value is fed in on every iteration.
625 * max_retry : upper bound on read attempts (>= 1). The first read
626 * counts as attempt #1.
627 * interval_ms : delay between consecutive reads. 0 means "re-issue
628 * immediately on completion" (use only if the bus already
629 * paces reads, e.g. USB ep0 vendor request).
630 * done : completion callback.
631 * VSF_ERR_NONE : predicate matched.
632 * VSF_ERR_TIMEOUT : retries exhausted without match.
633 * other : bus-level error.
634 *
635 * Same single-flight constraint as run_script / run_blob.
636 */
637vsf_err_t vsf_wifi_reg_read_poll(vsf_wifi_t *wifi, uint16_t reg,
638 vsf_wifi_reg_match_fn_t match,
639 uint16_t max_retry, uint16_t interval_ms,
640 vsf_wifi_done_t done);

642/*

647vsf_err_t vsf_wifi_reg_read(vsf_wifi_t *wifi, uint16_t reg, uint32_t *out,
