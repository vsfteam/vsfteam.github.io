70typedef enum vsf_usart_mode_t {
71 // 0..1: STOP(13:12) in USART_CR2, shift right by 12 to avoid conflict with BIT_LENGTH
72 VSF_USART_0_5_STOPBIT = (1 << 12) >> 12,
73 VSF_USART_1_STOPBIT = (0 << 12) >> 12,
74 VSF_USART_1_5_STOPBIT = (3 << 12) >> 12,
75 VSF_USART_2_STOPBIT = (2 << 12) >> 12,
76
77 // 2..3: RE(2)/TE(3) in USART_CR1
78 VSF_USART_TX_ENABLE = (1 << 3),
79 VSF_USART_TX_DISABLE = (0 << 3),
80 VSF_USART_RX_ENABLE = (1 << 2),
81 VSF_USART_RX_DISABLE = (0 << 2),
82
83 // 4..5: RTSE(8)/CTSE(9) in USART_CR3, shift right by 4 to void conflict with PARITY
84 VSF_USART_NO_HWCONTROL = (0),
85 VSF_USART_RTS_HWCONTROL = (1 << 8) >> 4,
86 VSF_USART_CTS_HWCONTROL = (1 << 9) >> 4,
87 VSF_USART_RTS_CTS_HWCONTROL = VSF_USART_RTS_HWCONTROL
88 | VSF_USART_CTS_HWCONTROL,
89
90 // 6: HDSEL(3) in USART_CR3, shift left by 3
91 VSF_USART_HALF_DUPLEX_ENABLE = (1 << 3) << 3,
92 VSF_USART_HALF_DUPLEX_DISABLE = 0,
93
94 // 9..10: PS(9)/PCE(10) in USART_CR1
95 VSF_USART_NO_PARITY = (0 << 10),
96 VSF_USART_ODD_PARITY = (1 << 10) | (1 << 9),
97 VSF_USART_EVEN_PARITY = (1 << 10),
98
99 // 11: CLKEN(11) in USART_CR2
100 VSF_USART_SYNC_CLOCK_ENABLE = (1 << 11),
101 VSF_USART_SYNC_CLOCK_DISABLE = (0 << 11),
102 // 7..8: CPOL(10)/CPHA(9) in USART_CR2, shift right by 2
103 VSF_USART_SYNC_CLOCK_POLARITY_LOW = (0 << 10) >> 2,
104 VSF_USART_SYNC_CLOCK_POLARITY_HIGH = (1 << 10) >> 2,
105 VSF_USART_SYNC_CLOCK_PHASE_1_EDGE = (0 << 9) >> 2,
106 VSF_USART_SYNC_CLOCK_PHASE_2_EDGE = (1 << 9) >> 2,
107
108 // 12&28: M0(12):M1(28) in USART_CR1
109 VSF_USART_9_BIT_LENGTH = (1 << 12),
110 VSF_USART_8_BIT_LENGTH = (0),
111 VSF_USART_7_BIT_LENGTH = (1 << 28),
112
113 // 25..27&29..31: TXFTCFG(29..31) & RXFTCFG(25..27)
114 VSF_USART_TX_FIFO_THRESHOLD_NOT_FULL = (0 << 29),
115 VSF_USART_TX_FIFO_THRESHOLD_HALF_EMPTY = (2 << 29),
116 VSF_USART_TX_FIFO_THRESHOLD_EMPTY = (5 << 29),
117 VSF_USART_RX_FIFO_THRESHOLD_NOT_EMPTY = (0 << 25),
118 VSF_USART_RX_FIFO_THRESHOLD_HALF_FULL = (2 << 25),
119 VSF_USART_RX_FIFO_THRESHOLD_FULL = (5 << 25),
120
121 // more vendor specified modes can be added here
122
123 // 15: SWAP(15) in USART_CR2
124 VSF_USART_SWAP = (1 << 15),
125 // 16..17: RXINV(16)/TXINV(17) in USART_CR2
126 VSF_USART_TX_INV = (1 << 17),
127 VSF_USART_RX_INV = (1 << 16),
128 // 18: OVER8(15) in USART_CR1, left shifted by 3 to avoid conflict with SWAP
129 VSF_USART_OVERSAMPLE_8 = (1 << 15) << 3,
130 VSF_USART_OVERSAMPLE_16 = (0 << 15) << 3,
131 VSF_USART_OVERSAMPLE_MASK = (1 << 15) << 3,
132
133 __VSF_HW_USART_MODE_CR1_MASK = VSF_USART_9_BIT_LENGTH
134 | VSF_USART_8_BIT_LENGTH
135 | VSF_USART_7_BIT_LENGTH
136 | VSF_USART_NO_PARITY
137 | VSF_USART_ODD_PARITY
138 | VSF_USART_EVEN_PARITY
139 | VSF_USART_TX_ENABLE
140 | VSF_USART_RX_ENABLE
141 | VSF_USART_OVERSAMPLE_MASK,
142 __VSF_HW_USART_MODE_CR2_MASK = VSF_USART_SYNC_CLOCK_ENABLE
143 | VSF_USART_SYNC_CLOCK_DISABLE
144 | VSF_USART_SYNC_CLOCK_POLARITY_HIGH
145 | VSF_USART_SYNC_CLOCK_POLARITY_LOW
146 | VSF_USART_SYNC_CLOCK_PHASE_1_EDGE
147 | VSF_USART_SYNC_CLOCK_PHASE_2_EDGE
148 | VSF_USART_0_5_STOPBIT
149 | VSF_USART_1_STOPBIT
150 | VSF_USART_1_5_STOPBIT
151 | VSF_USART_2_STOPBIT
152 | VSF_USART_TX_INV
153 | VSF_USART_RX_INV
154 | VSF_USART_SWAP,
155 __VSF_HW_USART_MODE_CR3_MASK = VSF_USART_RTS_CTS_HWCONTROL
156 | VSF_USART_HALF_DUPLEX_ENABLE,
157
158
159 // not supported, allocate unused bits
160 // 6..8
161 VSF_USART_6_BIT_LENGTH = (1 << 20),
162 VSF_USART_5_BIT_LENGTH = (2 << 20),
163 VSF_USART_10_BIT_LENGTH = (3 << 20),
164 VSF_USART_FORCE_0_PARITY = (0 << 8),
165 VSF_USART_FORCE_1_PARITY = (1 << 8),
166
167 __VSF_HW_USART_NOT_SUPPORT_MASK = VSF_USART_5_BIT_LENGTH
168 | VSF_USART_6_BIT_LENGTH
169 | VSF_USART_10_BIT_LENGTH
170 | VSF_USART_FORCE_0_PARITY
171 | VSF_USART_FORCE_1_PARITY,
172} vsf_usart_mode_t;
