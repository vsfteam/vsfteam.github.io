  SDIO_CMDOP_SINGLE_BLOCK = (0ul << 2) ,

  SDIO_CMDOP_WRITE = __SDIO_CMDOP_DTEN ,
  __SDIO_CMDOP_RESP = ((1ul << 12) << 2) ,
  __SDIO_CMDOP_RESP_NONE = ((0ul << 8) << 2) ,
