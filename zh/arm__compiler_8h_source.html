241 static ALWAYS_INLINE uint32_t ____disable_irq(void)
242 {
243  uint32_t cpsr;
244 
245  __asm__ __volatile__("mrs %[cpsr], primask\n"
246  "cpsid i\n"
247  : [cpsr] "=r"(cpsr));
248  return cpsr & 0x1;
249 }
