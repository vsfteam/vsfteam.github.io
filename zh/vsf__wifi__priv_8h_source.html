29/*
30 * The wifi layer is intentionally bus-agnostic, so vsf_wifi_t holds NO
31 * USB / SDIO / SPI specifics. The bus driver embeds vsf_wifi_t inside a
32 * larger struct (e.g. vk_usbh_wifi_t) and recovers its outer instance via
33 * vsf_container_of() inside the bus_ops implementations.
34 */
35struct vsf_wifi_t {
36 const vsf_wifi_chip_drv_t *drv;
37 const vsf_wifi_bus_ops_t *bus_ops;
38 vsf_eda_t *post_eda;
39 vsf_wifi_attach_fail_t attach_fail;
40
41 uint8_t channel;
42 bool is_ready;
43 bool disconnecting; /* set in fini() to gate stale callbacks */
44 uint8_t mac[6]; /* populated by chip-driver EEPROM stage; */
45 /* zero until firmware_load chain finishes */
46
47 /* ---- Scan state (wifi-driven hop scheduler) ---- */
48 bool scanning;
49 uint8_t scan_channel_idx;
50 uint8_t scan_num_channels;
51 uint8_t scan_channels[VSF_WIFI_CFG_SCAN_MAX_CHANNELS];
52 uint16_t scan_dwell_ms;
53#if VSF_KERNEL_CFG_SUPPORT_CALLBACK_TIMER == ENABLED
54 vsf_callback_timer_t scan_timer;
55 /* Read-poll spacing timer. Logically independent of scan_timer:
56 * read_poll runs during firmware_load (is_ready == false) while scan
57 * runs after on_ready, but they could in principle overlap if a
58 * future chip op uses run_read_poll mid-flight, so we keep two. */
59 vsf_callback_timer_t read_poll_timer;
60#endif
61
62 /* ---- MLME connection state (OPEN-system auth + association) ----
63 *
64 * Driven by the wifi layer (vsf_wifi.c). mlme_state walks
65 * IDLE -> AUTH -> ASSOC -> RUN; each handshake step arms mlme_timer and
66 * retransmits up to mlme_retry times before giving up. */
67 uint8_t mlme_state; /* WIFI_MLME_xxx */
68 uint8_t mlme_bssid[6]; /* target AP BSSID */
69 uint8_t mlme_ssid[33]; /* target SSID (for assoc-req SSID IE) */
70 uint8_t mlme_ssid_len;
71 uint8_t mlme_channel; /* target channel */
72 uint8_t mlme_retry; /* retransmits left for current step */
73 uint16_t mlme_aid; /* association id from assoc-resp */
74#if VSF_KERNEL_CFG_SUPPORT_CALLBACK_TIMER == ENABLED
75 vsf_callback_timer_t mlme_timer;
76#endif
77
78#if VSF_WIFI_USE_WPA == ENABLED
79 /* ---- WPA2-PSK security context ----
80 *
81 * wpa_auth is captured by vsf_wifi_set_auth_mode(); auth_mode == OPEN
82 * means no RSN (the assoc-req carries no RSN IE and no 4-way handshake
83 * runs). wpa_rsn_ie holds the STA RSN IE bytes sent in the assoc-req so
84 * the 4-way handshake M2 can echo them unchanged. Handshake-specific
85 * fields (PTK / nonces / GTK / replay / substate) are added by the WPA
86 * state machine. */
87 vsf_wifi_auth_cfg_t wpa_auth;
88 uint8_t wpa_rsn_ie[24];
89 uint8_t wpa_rsn_ie_len;
90
91 /* 4-way handshake working state (vsf_wifi_wpa.c). The PMK lives in
92 * wpa_auth.psk; PTK = KCK(16)|KEK(16)|TK(16) is derived on M1. anonce /
93 * snonce feed the PRF; gtk holds the group key unwrapped from M3.
94 * replay echoes the AP's EAPOL-Key replay counter; ptk_valid gates the
95 * CCMP data path until the handshake installs keys. */
96 uint8_t wpa_ptk[48];
97 uint8_t wpa_anonce[32];
98 uint8_t wpa_snonce[32];
99 uint8_t wpa_gtk[32];
100 uint8_t wpa_gtk_len;
101 uint8_t wpa_gtk_keyidx;
102 uint8_t wpa_replay[8];
103 bool wpa_ptk_valid;
104
105 /* CCMP data path (vsf_wifi.c). wpa_tx_pn is the 48-bit packet-number
106 * counter (little-endian, pn[0] = LSB) advanced once per encrypted TX
107 * MPDU; it seeds the CCMP header and CCM nonce. wpa_hw_crypto is set
108 * when a chip crypto_ops backend installed the keys, in which case the
109 * software CCMP encap/decap is bypassed (the chip does it in-line). */
110 uint8_t wpa_tx_pn[6];
111 bool wpa_hw_crypto;
112#endif
113
114 /* ---- Script / blob dispatcher state ----
115 *
116 * Only one outstanding script or blob is allowed per wifi (the bus_ops
117 * concurrency contract enforces the same limit at the bus level). The
118 * dispatcher walks the op array via the chained __script_step_done /
119 * __blob_step_done callbacks, each of which feeds the next op to
120 * bus_ops.
121 */
122 bool script_busy;
123 bool script_is_blob;
124 vsf_wifi_done_t script_done;
125 union {
126 struct {
127 const vsf_wifi_op_t *ops;
128 uint16_t count;
129 uint16_t idx;
130 } script;
131 struct {
132 const uint8_t *data;
133 uint32_t len;
134 uint32_t offset; /* used by the reg_write fallback path */
135 uint16_t base_reg;
136 uint16_t chunk_size;
137 } blob;
138 struct {
139 uint16_t reg;
140 uint16_t retry_left; /* attempts remaining after
141 * the current in-flight read */
142 uint16_t interval_ms;
143 uint16_t reserved;
144 uint32_t last_val; /* read landing zone */
145 vsf_wifi_match_fn_t match;
146 } read_poll;
147 } s;
148
149 /*
150 * Backend-level chain slot. When a chip op needs to issue several
151 * scripts back-to-back (firmware_load = run_blob + post-fw script),
152 * it stashes the *outer* `done` here, then internally chains via local
153 * helpers. Only one chain may be in flight at a time — same constraint
154 * as the script dispatcher itself.
155 */
156 vsf_wifi_done_t backend_chain_done;
157
158 vsf_wifi_op_t scratch_ops[VSF_WIFI_CFG_SCRATCH_OPS];
159};
