29/*
30 * The wifi layer is intentionally bus-agnostic, so vsf_wifi_t holds NO
31 * USB / SDIO / SPI specifics. The bus driver embeds vsf_wifi_t inside a
32 * larger struct (e.g. vk_usbh_wifi_t) and recovers its outer instance via
33 * vsf_container_of() inside the bus_ops implementations.
34 */
35struct vsf_wifi_t {
36 const vsf_wifi_chip_drv_t *drv;
37 const vsf_wifi_bus_ops_t *bus_ops;
38 vsf_eda_t *post_eda;
39 vsf_wifi_attach_fail_t attach_fail;
40
41 uint8_t channel;
42 bool is_ready;
43 bool disconnecting; /* set in fini() to gate stale callbacks */
44 uint8_t mac[6]; /* populated by chip-driver EEPROM stage; */
45 /* zero until firmware_load chain finishes */
46
47 /* ---- Scan state (wifi-driven hop scheduler) ---- */
48 bool scanning;
49 uint8_t scan_channel_idx;
50 uint8_t scan_num_channels;
51 uint8_t scan_channels[VSF_WIFI_CFG_SCAN_MAX_CHANNELS];
52 uint16_t scan_dwell_ms;
53#if VSF_KERNEL_CFG_SUPPORT_CALLBACK_TIMER == ENABLED
54 vsf_callback_timer_t scan_timer;
55 /* Read-poll spacing timer. Logically independent of scan_timer:
56 * read_poll runs during firmware_load (is_ready == false) while scan
57 * runs after on_ready, but they could in principle overlap if a
58 * future chip op uses run_read_poll mid-flight, so we keep two. */
59 vsf_callback_timer_t read_poll_timer;
60#endif
61
62 /* ---- Script / blob dispatcher state ----
63 *
64 * Only one outstanding script or blob is allowed per wifi (the bus_ops
65 * concurrency contract enforces the same limit at the bus level). The
66 * dispatcher walks the op array via the chained __script_step_done /
67 * __blob_step_done callbacks, each of which feeds the next op to
68 * bus_ops.
69 */
70 bool script_busy;
71 bool script_is_blob;
72 vsf_wifi_done_t script_done;
73 union {
74 struct {
75 const vsf_wifi_op_t *ops;
76 uint16_t count;
77 uint16_t idx;
78 } script;
79 struct {
80 const uint8_t *data;
81 uint32_t len;
82 uint32_t offset; /* used by the reg_write fallback path */
83 uint16_t base_reg;
84 uint16_t chunk_size;
85 } blob;
86 struct {
87 uint16_t reg;
88 uint16_t retry_left; /* attempts remaining after
89 * the current in-flight read */
90 uint16_t interval_ms;
91 uint16_t reserved;
92 uint32_t last_val; /* read landing zone */
93 vsf_wifi_match_fn_t match;
94 } read_poll;
95 } s;
96
97 /*
98 * Backend-level chain slot. When a chip op needs to issue several
99 * scripts back-to-back (firmware_load = run_blob + post-fw script),
100 * it stashes the *outer* `done` here, then internally chains via local
101 * helpers. Only one chain may be in flight at a time — same constraint
102 * as the script dispatcher itself.
103 */
104 vsf_wifi_done_t backend_chain_done;
105
106 vsf_wifi_op_t scratch_ops[VSF_WIFI_CFG_SCRATCH_OPS];
107};
