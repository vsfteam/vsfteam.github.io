29/*
30 * The wifi layer is chip-agnostic but knows about the optional register-bus
31 * helper used by register-based chips (e.g. RT2X00). reg_bus may be NULL for
32 * chips that use a command/event-based bus (e.g. MediaTek mt76). The bus
33 * driver embeds vsf_wifi_t inside a larger struct (e.g. vk_usbh_wifi_t) and
34 * recovers its outer instance via vsf_container_of().
35 */
36struct vsf_wifi_t {
37 const vsf_wifi_chip_drv_t *drv;
38 const vsf_wifi_reg_bus_t *reg_bus;
39 vsf_eda_t *post_eda;
40 vsf_wifi_attach_fail_t attach_fail;
41
42 /* Optional network-stack backend (lwIP netdrv adapter, etc.) bound at
43 * runtime via vsf_wifi_netdrv_attach(). When netif_ops != NULL the core
44 * routes decrypted RX data frames and link up/down events here instead
45 * of the weak vsf_wifi_on_* hooks. */
46 const vsf_wifi_netif_ops_t *netif_ops;
47 void *netif_param;
48
49 uint8_t channel;
50 bool is_ready;
51 bool disconnecting; /* set in fini() to gate stale callbacks */
52 uint8_t mac[6]; /* populated by chip-driver EEPROM stage; */
53 /* zero until firmware_load chain finishes */
54
55 /* ---- Scan state (wifi-driven hop scheduler) ---- */
56 bool scanning;
57 uint8_t scan_channel_idx;
58 uint8_t scan_num_channels;
59 uint8_t scan_channels[VSF_WIFI_CFG_SCAN_MAX_CHANNELS];
60 uint16_t scan_dwell_ms;
61#if VSF_KERNEL_CFG_SUPPORT_CALLBACK_TIMER == ENABLED
62 vsf_callback_timer_t scan_timer;
63 /* Read-poll spacing timer. Logically independent of scan_timer:
64 * read_poll runs during firmware_load (is_ready == false) while scan
65 * runs after on_ready, but they could in principle overlap if a
66 * future chip op uses run_read_poll mid-flight, so we keep two. */
67 vsf_callback_timer_t read_poll_timer;
68#endif
69
70 /* ---- MLME connection state (OPEN-system auth + association) ----
71 *
72 * Driven by the wifi layer (vsf_wifi.c). mlme_state walks
73 * IDLE -> AUTH -> ASSOC -> RUN; each handshake step arms mlme_timer and
74 * retransmits up to mlme_retry times before giving up. */
75 uint8_t mlme_state; /* WIFI_MLME_xxx */
76 uint8_t mlme_bssid[6]; /* target AP BSSID */
77 uint8_t mlme_ssid[33]; /* target SSID (for assoc-req SSID IE) */
78 uint8_t mlme_ssid_len;
79 uint8_t mlme_channel; /* target channel */
80 uint8_t mlme_retry; /* retransmits left for current step */
81 uint16_t mlme_aid; /* association id from assoc-resp */
82#if VSF_KERNEL_CFG_SUPPORT_CALLBACK_TIMER == ENABLED
83 vsf_callback_timer_t mlme_timer;
84#endif
85
86#if VSF_WIFI_USE_WPA == ENABLED
87 /* ---- WPA2-PSK security context ----
88 *
89 * wpa_auth is captured by vsf_wifi_set_auth_mode(); auth_mode == OPEN
90 * means no RSN (the assoc-req carries no RSN IE and no 4-way handshake
91 * runs). wpa_rsn_ie holds the STA RSN IE bytes sent in the assoc-req so
92 * the 4-way handshake M2 can echo them unchanged. Handshake-specific
93 * fields (PTK / nonces / GTK / replay / substate) are added by the WPA
94 * state machine. */
95 vsf_wifi_auth_cfg_t wpa_auth;
96 uint8_t wpa_rsn_ie[24];
97 uint8_t wpa_rsn_ie_len;
98
99 /* 4-way handshake working state (vsf_wifi_wpa.c). The PMK lives in
100 * wpa_auth.psk; PTK = KCK(16)|KEK(16)|TK(16) is derived on M1. anonce /
101 * snonce feed the PRF; gtk holds the group key unwrapped from M3.
102 * replay echoes the AP's EAPOL-Key replay counter; ptk_valid gates the
103 * CCMP data path until the handshake installs keys. */
104 uint8_t wpa_ptk[48];
105 uint8_t wpa_anonce[32];
106 uint8_t wpa_snonce[32];
107 uint8_t wpa_gtk[32];
108 uint8_t wpa_gtk_len;
109 uint8_t wpa_gtk_keyidx;
110 uint8_t wpa_replay[8];
111 bool wpa_ptk_valid;
112
113 /* CCMP data path (vsf_wifi.c). wpa_tx_pn is the 48-bit packet-number
114 * counter (little-endian, pn[0] = LSB) advanced once per encrypted TX
115 * MPDU; it seeds the CCMP header and CCM nonce. wpa_hw_crypto is set
116 * when a chip crypto_ops backend installed the keys, in which case the
117 * software CCMP encap/decap is bypassed (the chip does it in-line). */
118 uint8_t wpa_tx_pn[6];
119 bool wpa_hw_crypto;
120#endif
121
122 /* ---- Script / blob dispatcher state ----
123 *
124 * Only one outstanding script or blob is allowed per wifi (the reg_bus
125 * concurrency contract enforces the same limit at the bus level). The
126 * dispatcher walks the op array via the chained __script_step_done /
127 * __blob_step_done callbacks, each of which feeds the next op to
128 * reg_bus.
129 */
130 bool script_busy;
131 bool script_is_blob;
132 vsf_wifi_done_t script_done;
133 union {
134 struct {
135 const vsf_wifi_reg_op_t *ops;
136 uint16_t count;
137 uint16_t idx;
138 } script;
139 struct {
140 const uint8_t *data;
141 uint32_t len;
142 uint32_t offset; /* used by the reg_write fallback path */
143 uint16_t base_reg;
144 uint16_t chunk_size;
145 } blob;
146 struct {
147 uint16_t reg;
148 uint16_t retry_left; /* attempts remaining after
149 * the current in-flight read */
150 uint16_t interval_ms;
151 uint16_t reserved;
152 uint32_t last_val; /* read landing zone */
153 vsf_wifi_reg_match_fn_t match;
154 } read_poll;
155 } s;
156
157 /*
158 * Backend-level chain slot. When a chip op needs to issue several
159 * scripts back-to-back (firmware_load = run_blob + post-fw script),
160 * it stashes the *outer* `done` here, then internally chains via local
161 * helpers. Only one chain may be in flight at a time — same constraint
162 * as the script dispatcher itself.
163 */
164 vsf_wifi_done_t backend_chain_done;
165
166 vsf_wifi_reg_op_t scratch_ops[VSF_WIFI_CFG_SCRATCH_OPS];
167};

169#endif // VSF_USE_WIFI
