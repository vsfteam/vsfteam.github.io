41/*============================ TYPES =========================================*/
42
43/* Opaque per-interface handle. The concrete layout lives in the port. */
44typedef struct esp_netif_obj esp_netif_t;
45
46/* I/O driver handle passed to esp_netif_attach(). The port casts this to
47 * an esp_netif_driver_base_t * and, if post_attach is set, invokes it so
48 * the driver can call esp_netif_set_driver_config(). */
49typedef void * esp_netif_iodriver_handle;
50
51/* IPv4 configuration snapshot. All fields in network byte order. */
52typedef struct {
53 esp_ip4_addr_t ip;
54 esp_ip4_addr_t netmask;
55 esp_ip4_addr_t gw;
56} esp_netif_ip_info_t;
57
58/* IPv6 configuration snapshot. */
59typedef struct {
60 esp_ip6_addr_t ip;
61} esp_netif_ip6_info_t;
62
63/* DNS server record. */
64typedef struct {
65 esp_ip_addr_t ip;
66} esp_netif_dns_info_t;
67
68/* Which DNS slot to read or write. */
69typedef enum {
70 ESP_NETIF_DNS_MAIN = 0,
71 ESP_NETIF_DNS_BACKUP,
72 ESP_NETIF_DNS_FALLBACK,
73 ESP_NETIF_DNS_MAX
74} esp_netif_dns_type_t;
75
76/* DHCP client / server state. */
77typedef enum {
78 ESP_NETIF_DHCP_INIT = 0,
79 ESP_NETIF_DHCP_STARTED,
80 ESP_NETIF_DHCP_STOPPED,
81 ESP_NETIF_DHCP_STATUS_MAX
82} esp_netif_dhcp_status_t;
83
84/* Bitmask of per-interface feature flags (esp_netif_inherent_config_t). */
85typedef enum {
86 ESP_NETIF_DHCP_CLIENT = 1 << 0,
87 ESP_NETIF_DHCP_SERVER = 1 << 1,
88 ESP_NETIF_FLAG_AUTOUP = 1 << 2,
89 ESP_NETIF_FLAG_GARP = 1 << 3,
90 ESP_NETIF_FLAG_EVENT_IP_MODIFIED = 1 << 4,
91 ESP_NETIF_FLAG_IS_PPP = 1 << 5,
92 ESP_NETIF_FLAG_IS_BRIDGE = 1 << 6,
93 ESP_NETIF_FLAG_MLDV6_REPORT = 1 << 7,
94} esp_netif_flags_t;
95
96/* Initial L3 configuration applied when the interface is created. */
97typedef struct esp_netif_inherent_config {
98 esp_netif_flags_t flags;
99 uint8_t mac[6];
100 const esp_netif_ip_info_t *ip_info;
101 uint32_t get_ip_event;
102 uint32_t lost_ip_event;
103 const char * if_key;
104 const char * if_desc;
105 int route_prio;
106 uint32_t bridge_info;
107} esp_netif_inherent_config_t;
108
109/* I/O driver contract passed back through esp_netif_driver_base_t::post_attach.
110 * transmit / driver_free_rx_buffer are invoked by esp_netif when the stack
111 * wants to send a frame or release a received buffer. */
112typedef struct esp_netif_driver_ifconfig {
113 esp_netif_iodriver_handle handle;
114 esp_err_t (*transmit)(void *h, void *buffer, size_t len);
115 esp_err_t (*transmit_wrap)(void *h, void *buffer, size_t len, void *netstack_buf);
116 void (*driver_free_rx_buffer)(void *h, void *buffer);
117} esp_netif_driver_ifconfig_t;
118
119/* Driver-side base object. Users typedef their own struct with this as the
120 * first member so esp_netif_attach() can downcast and call post_attach. */
121typedef struct esp_netif_driver_base {
122 esp_err_t (*post_attach)(esp_netif_t *netif, esp_netif_iodriver_handle h);
123 esp_netif_t *netif;
124} esp_netif_driver_base_t;
125
126/* Layered config accepted by esp_netif_new(). stack_config and driver_config
127 * are optional; a zero-filled struct is enough when the port supplies its
128 * own defaults (see esp_netif_defaults.h). */
129typedef struct esp_netif_config {
130 const esp_netif_inherent_config_t *base;
131 const void *driver_config;
132 const void *stack_config;
133} esp_netif_config_t;

137/* Event ids dispatched through the process-wide default event loop on the
138 * IP_EVENT base. Values match the ESP-IDF v5.x public contract. */
139typedef enum {
140 IP_EVENT_STA_GOT_IP = 0,
141 IP_EVENT_STA_LOST_IP,
142 IP_EVENT_AP_STAIPASSIGNED,
143 IP_EVENT_GOT_IP6,
144 IP_EVENT_ETH_GOT_IP,
145 IP_EVENT_ETH_LOST_IP,
146 IP_EVENT_PPP_GOT_IP,
147 IP_EVENT_PPP_LOST_IP,
148} ip_event_t;
149
150/* Payload dispatched with IP_EVENT_STA_GOT_IP / IP_EVENT_ETH_GOT_IP. */
151typedef struct {
152 esp_netif_t *esp_netif;
153 esp_netif_ip_info_t ip_info;
154 bool ip_changed;
155} ip_event_got_ip_t;
156
157/* Payload dispatched with IP_EVENT_GOT_IP6. */
158typedef struct {
159 esp_netif_t *esp_netif;
160 esp_netif_ip6_info_t ip6_info;
161 int ip_index;
162} ip_event_got_ip6_t;
163
164/* Payload dispatched with IP_EVENT_AP_STAIPASSIGNED. */
165typedef struct {
166 esp_netif_t *esp_netif;
167 esp_ip4_addr_t ip;
168 uint8_t mac[6];
169} ip_event_ap_staipassigned_t;
