45 // bit0 - 7: index
46 // bit8 - 12: bitlen - 1
47 // bit13 - 14: bytelen - 1
48 // bit15: has_alpha
49 #define VSF_DISP_COLOR_VALUE(__INDEX, __BITLEN, __BYTELEN, __HAS_ALPHA) \
50  ( (__INDEX) \
51  | ((((__BITLEN) - 1) & 0x1F) << 8) \
52  | ((((__BYTELEN) - 1) & 0x03) << 13) \
53  | ((__HAS_ALPHA) ? 1 << 15 : 0) \
54  )
55 
56 #define VSF_DISP_COLOR_DEF(__NAME, __BITLEN, __BYTELEN, __HAS_ALPHA) \
57  VSF_DISP_COLOR_##__NAME = VSF_DISP_COLOR_VALUE( \
58  (VSF_DISP_COLOR_IDX_##__NAME), __BITLEN, __BYTELEN, __HAS_ALPHA)
