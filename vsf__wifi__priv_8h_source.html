46/*
47 * The wifi layer is chip-agnostic but knows about the optional register-bus
48 * helper used by register-based chips (e.g. RT2X00). reg_bus may be NULL for
49 * chips that use a command/event-based bus (e.g. MediaTek mt76). The bus
50 * driver embeds vsf_wifi_t inside a larger struct (e.g. vk_usbh_wifi_t) and
51 * recovers its outer instance via vsf_container_of().
52 */

62 * of the weak vsf_wifi_on_* hooks. */

64 void *netif_param;

66 uint8_t channel;
67 bool is_ready;
68 bool disconnecting; /* set in fini() to gate stale callbacks */

70 /* zero until firmware_load chain finishes */

87 /* ---- MLME connection state (OPEN-system auth + association) ----
88 *
89 * Driven by the wifi layer (vsf_wifi.c). mlme_state walks

91 * retransmits up to mlme_retry times before giving up. */

103#if VSF_WIFI_USE_WPA == ENABLED
104 /* ---- WPA2-PSK security context ----
105 *
106 * wpa_auth is captured by vsf_wifi_set_auth_mode(); auth_mode == OPEN
107 * means no RSN (the assoc-req carries no RSN IE and no 4-way handshake
108 * runs). wpa_rsn_ie holds the STA RSN IE bytes sent in the assoc-req so
109 * the 4-way handshake M2 can echo them unchanged. Handshake-specific
110 * fields (PTK / nonces / GTK / replay / substate) are added by the WPA
111 * state machine. */
112 vsf_wifi_auth_cfg_t wpa_auth;
113 uint8_t wpa_rsn_ie[24];
114 uint8_t wpa_rsn_ie_len;
115
116 /* 4-way handshake working state (vsf_wifi_wpa.c). The PMK lives in
117 * wpa_auth.psk; PTK = KCK(16)|KEK(16)|TK(16) is derived on M1. anonce /
118 * snonce feed the PRF; gtk holds the group key unwrapped from M3.
119 * replay echoes the AP's EAPOL-Key replay counter; ptk_valid gates the
120 * CCMP data path until the handshake installs keys. */
121 uint8_t wpa_ptk[48];
122 uint8_t wpa_anonce[32];
123 uint8_t wpa_snonce[32];
124 uint8_t wpa_gtk[32];
125 uint8_t wpa_gtk_len;
126 uint8_t wpa_gtk_keyidx;
127 uint8_t wpa_replay[8];
128 bool wpa_ptk_valid;
129
130 /* CCMP data path (vsf_wifi.c). wpa_tx_pn is the 48-bit packet-number
131 * counter (little-endian, pn[0] = LSB) advanced once per encrypted TX
132 * MPDU; it seeds the CCMP header and CCM nonce. wpa_hw_crypto is set
133 * when a chip crypto_ops backend installed the keys, in which case the
134 * software CCMP encap/decap is bypassed (the chip does it in-line). */
135 uint8_t wpa_tx_pn[6];
136 bool wpa_hw_crypto;
137#endif

139 /* ---- Script / blob dispatcher state ----
140 *
141 * Only one outstanding script or blob is allowed per wifi (the reg_bus
142 * concurrency contract enforces the same limit at the bus level). The
143 * dispatcher walks the op array via the chained __script_step_done /
144 * __blob_step_done callbacks, each of which feeds the next op to
145 * reg_bus.
146 */
147 bool script_busy;
148 bool script_is_blob;
149 vsf_wifi_done_t script_done;
150 union {
151 struct {
152 const vsf_wifi_reg_op_t *ops;
153 uint16_t count;
154 uint16_t idx;
155 } script;
156 struct {
157 const uint8_t *data;
158 uint32_t len;
159 uint32_t offset; /* used by the reg_write fallback path */
160 uint16_t base_reg;
161 uint16_t chunk_size;
162 } blob;
163 struct {
164 uint16_t reg;
165 uint16_t retry_left; /* attempts remaining after
166 * the current in-flight read */
167 uint16_t interval_ms;
168 uint16_t reserved;
169 uint32_t last_val; /* read landing zone */
170 vsf_wifi_reg_match_fn_t match;
171 } read_poll;
172 } s;
173
174 /*
175 * Backend-level chain slot. When a chip op needs to issue several
176 * scripts back-to-back (firmware_load = run_blob + post-fw script),
177 * it stashes the *outer* `done` here, then internally chains via local
178 * helpers. Only one chain may be in flight at a time — same constraint
179 * as the script dispatcher itself.
180 */
181 vsf_wifi_done_t backend_chain_done;
182
183 /* BSS Max Idle Period from assoc-resp IE 90 (units of 1000 TU).
184 * 0 means the AP did not advertise it. */
185 uint16_t bss_max_idle_period;
186
187#if (VSF_WIFI_CFG_KEEPALIVE_PERIOD_MS > 0) \
188 && (VSF_KERNEL_CFG_SUPPORT_CALLBACK_TIMER == ENABLED)
189 /* 802.11 null-data-frame keepalive state. Started on link-up and stopped
190 * on link-down / fini to keep the AP from deauthenticating an idle STA. */
191 vsf_callback_timer_t keepalive_timer;
192 uint32_t keepalive_period_ms;
193#endif
194
195 vsf_wifi_reg_op_t scratch_ops[VSF_WIFI_CFG_SCRATCH_OPS];
196
197 /* Chip-driver private data pointer. Set by the bus driver after
198 * vsf_wifi_init() for command/event-based chips that need per-instance
199 * state beyond the generic wifi struct (e.g. MediaTek mt76). The generic
200 * layer never touches this field. */
201 void *chip_priv;
202
203 /* Chip-specific bus operations.
204 *
205 * This abstracts the *communication semantics* between the chip driver and
206 * the host, not the underlying physical bus. Register-based chips (e.g.
207 * Ralink RT28xx) use reg_bus above: the chip driver only knows "read/write
208 * a register", and the bus driver implements that with USB ep0 vendor
209 * requests, SDIO CMD53, etc.
210 *
211 * Command/event-based chips (e.g. MediaTek mt76) store a chip-defined
212 * vtable here (cast by the chip driver to the appropriate type). The
213 * operations are named after what the *chip* does (send MCU command,
214 * submit TX frame, start RX, ...), so the same chip driver can plug into
215 * different physical buses without modification. */
216 const void *bus_ops;
217
218 /* ---- Raw WiFi radio state ----
219 *
220 * When raw_radio_active is true the instance is in raw-radio mode and the
221 * standard MLME/scan/WPA state machine is gated out. Received frames are
222 * delivered to raw_radio.rx_cb instead of the normal netif/weak hooks. */
223 bool raw_radio_active;
224 vsf_wifi_radio_t raw_radio;
225};
