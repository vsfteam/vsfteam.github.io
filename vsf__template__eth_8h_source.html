387 /* Address Filtering Options
388 * ------------------------- */
389 /*
390 // \~english Enable filtering of unicast frames not addressed to this station
391 // \~chinese 启用单播帧过滤，丢弃不是发往本站的单播帧
392 VSF_ETH_MODE_UNICAST_FILTER = 1 << 7,

394 // \~english Enable filtering of multicast frames according to a filter table
395 // \~chinese 启用多播帧过滤，根据过滤表过滤多播帧

398 // \~english Enable filtering of broadcast frames
399 // \~chinese 启用广播帧过滤，可选择是否接收广播帧
400 VSF_ETH_MODE_BROADCAST_FILTER = 1 << 9,
401 */
402
403 /* VLAN Support Options
404 * --------------------
405 * VLAN (Virtual LAN) allows network segmentation at Layer 2
406 */
407 /*
408 // \~english Enable adding VLAN tags to outgoing frames
409 // \~chinese 启用VLAN标签，向外发送的帧添加VLAN标签
410 VSF_ETH_MODE_VLAN_TAGGING = 1 << 10,
411
412 // \~english Enable filtering frames based on VLAN ID
413 // \~chinese 启用VLAN标签过滤，根据VLAN ID过滤接收的帧
414 VSF_ETH_MODE_VLAN_FILTERING = 1 << 11,
415
416 // \~english Remove VLAN tags from received frames before passing to upper layer
417 // \~chinese 启用VLAN标签剥离，在传递给上层前移除接收帧中的VLAN标签
418 VSF_ETH_MODE_VLAN_STRIPPING = 1 << 12,
419
420 // \~english Insert VLAN tags into outgoing frames that don't have them
421 // \~chinese 启用VLAN标签插入，向没有VLAN标签的外发帧插入标签
422 VSF_ETH_MODE_VLAN_INSERTION = 1 << 13,
423 */
424
425 /* Special Frame Handling
426 * --------------------- */
427 /*
428 // \~english Support for frames larger than 1518 bytes (up to 9000 bytes typically)
429 // \~chinese 支持巨型帧，允许处理大于1518字节的帧(通常最大可达9000字节)
430 VSF_ETH_MODE_JUMBO_FRAME = 1 << 14,

432 // \~english Support IEEE 802.3x flow control mechanism to prevent buffer overflow

435
436 // \~english Support for pause frames which temporarily pause transmission
437 // \~chinese 支持暂停帧，可临时暂停传输以控制数据流
438 VSF_ETH_MODE_PAUSE_FRAME = 1 << 16,
439 */
440
441 /* QoS and Traffic Management
442 * ------------------------- */
443 /*
444 // \~english Support for multiple priority queues for traffic handling
445 // \~chinese 支持优先级队列，根据优先级处理不同流量

453 /* Time Synchronization

456 // \~english Support for Precision Time Protocol timestamp for accurate timing

460
461 /* Advanced Hardware Acceleration

464 // \~english TCP Segmentation Offload - hardware handles TCP segmentation
